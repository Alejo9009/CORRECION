CREATE PROCEDURE EditarPiso(
    IN p_idPiso INT,
    IN p_nombrePiso VARCHAR(140)
)
BEGIN
    UPDATE piso
    SET nombrePiso = p_nombrePiso
    WHERE idPiso = p_idPiso;
END //
